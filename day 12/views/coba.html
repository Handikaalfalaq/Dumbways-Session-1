package main import ( "fmt" "time" ) func main() { // Mendefinisikan layout format waktu layout := "2006-01-02" // Mendapatkan input tanggal dari pengguna input1 := "2023-04-20" input2 := "2023-04-13" // Mengubah input menjadi tipe data time.Time date1, err := time.Parse(layout, input1) if err != nil { fmt.Println("Format input tanggal 1 salah") return } date2, err := time.Parse(layout, input2) if err != nil { fmt.Println("Format input tanggal 2 salah") return } // Menghitung selisih antara dua tanggal diff := date1.Sub(date2) // Menampilkan selisih dalam satuan yang diinginkan (dalam contoh ini, hari) fmt.Println("Selisih antara kedua tanggal adalah", int(diff.Hours()/24), "hari") }
